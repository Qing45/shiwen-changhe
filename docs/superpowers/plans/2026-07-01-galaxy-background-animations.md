```tsx
// Nebula clouds: large blurred colored blobs that drift slowly across the
// canvas. Each is wrapped so the outer div centers via translate(-50%, -50%)
// while the inner div owns the drift animation — keeping the two transforms
// from clobbering each other.
const NEBULA_CLOUDS = [
  { x: 18, y: 50, w: 700, h: 240, color: 'rgba(180,140,220,0.20)', dur: 50, delay: 0 },
  { x: 50, y: 45, w: 820, h: 280, color: 'rgba(150,180,230,0.18)', dur: 62, delay: -15 },
  { x: 78, y: 52, w: 600, h: 220, color: 'rgba(220,160,180,0.16)', dur: 55, delay: -25 },
  { x: 35, y: 55, w: 520, h: 180, color: 'rgba(200,180,140,0.14)', dur: 48, delay: -8 },
];

// 60 stars; 60% biased into the Milky Way band (top 28-72%), rest scattered
// across the full canvas. Sizes/durations/delays all randomized so no two
// twinkle in phase.
const STARS = Array.from({ length: 60 }, () => {
  const inBand = Math.random() < 0.6;
  return {
    top: inBand ? 28 + Math.random() * 44 : Math.random() * 100,
    left: Math.random() * 100,
    size: 0.6 + Math.random() * 1.4,
    duration: 2 + Math.random() * 4,
    delay: -Math.random() * 6,
  };
});

// 10 shooting stars, each with random start, angle (25-35°), travel distance,
// duration, and delay. Negative delays spread them across time so they never
// all fire together.
const SHOOTING_STARS = Array.from({ length: 10 }, () => {
  const angle = 25 + Math.random() * 10;
  const rad = (angle * Math.PI) / 180;
  const distance = 300 + Math.random() * 200;
  return {
    top: Math.random() * 30,
    left: 5 + Math.random() * 60,
    angle,
    dx: Math.cos(rad) * distance,
    dy: Math.sin(rad) * distance,
    duration: 1.0 + Math.random() * 0.6,
    delay: -4 - Math.random() * 10,
    trailLength: 80 + Math.random() * 40,
  };
});

// Parallax coefficients: max pixel offset per layer at mouse extremes (±1).
// Moon moves most (furthest perceived depth), stars least (nearest).
const PARALLAX_COEFS = {
  moon: 12,
  milkyWay: 8.4,
  nebula: 6,
  stars: 3.6,
};
```

**Notes:**
- `NEBULA_CLOUDS` and `STARS` are preserved byte-for-byte from the existing file
- `SHOOTING_STARS` is `Array.from` so the random values are computed once at module load (stable across re-renders, otherwise stars would teleport on every state change)
- `PARALLAX_COEFS` values come from the spec's depth table

Replace the existing `export function RiverBackground() { ... }` (the entire function body) with:

```tsx
export function RiverBackground({ dragging }: { dragging: boolean }) {
  const [mouse, setMouse] = useState({ x: 0, y: 0 });
  const mouseRef = useRef({ x: 0, y: 0 });
  const rafRef = useRef<number | null>(null);
  const draggingRef = useRef(dragging);

  // Keep draggingRef in sync so the mousemove handler (registered once) reads
  // current value without re-binding on every state change.
  useEffect(() => { draggingRef.current = dragging; }, [dragging]);

  // Listen at window level so parallax works even when the cursor is over a
  // node label, not just empty canvas. rAF throttle caps updates at one per
  // frame; drag freezes parallax entirely to avoid fighting the pan.
  useEffect(() => {
    const handler = (e: MouseEvent) => {
      if (draggingRef.current) return;
      mouseRef.current = {
        x: (e.clientX / window.innerWidth - 0.5) * 2,
        y: (e.clientY / window.innerHeight - 0.5) * 2,
      };
      if (rafRef.current == null) {
        rafRef.current = requestAnimationFrame(() => {
          rafRef.current = null;
          setMouse(mouseRef.current);
        });
      }
    };
    window.addEventListener('mousemove', handler);
    return () => {
      window.removeEventListener('mousemove', handler);
      if (rafRef.current != null) cancelAnimationFrame(rafRef.current);
    };
  }, []);

  const parallax = (coef: number) =>
    `translate(${(mouse.x * coef).toFixed(1)}px, ${(mouse.y * coef).toFixed(1)}px)`;

  return (
    <>
      {/* 星云气尘 — parallax wrapper around existing clouds */}
      <div style={{
        position: 'absolute', inset: 0,
        transform: parallax(PARALLAX_COEFS.nebula),
        willChange: 'transform',
        pointerEvents: 'none',
      }}>
        {NEBULA_CLOUDS.map((c, i) => (
          <div key={i} style={{
            position: 'absolute',
            top: `${c.y}%`,
            left: `${c.x}%`,
            transform: 'translate(-50%, -50%)',
          }}>
            <div style={{
              width: c.w,
              height: c.h,
              background: `radial-gradient(ellipse, ${c.color} 0%, transparent 70%)`,
              filter: 'blur(40px)',
              animation: `nebula-drift ${c.dur}s ease-in-out ${c.delay}s infinite alternate`,
            }} />
          </div>
        ))}
      </div>

      {/* 银河主带 — flows horizontally (120s) + parallax translate */}
      <div style={{
        position: 'absolute', top: '50%', left: 0, right: 0, height: '50%',
        transform: `translateY(-50%) ${parallax(PARALLAX_COEFS.milkyWay)}`,
        background: `
          linear-gradient(180deg,
            transparent 0%,
            rgba(180,200,240,0.04) 25%,
            rgba(216,224,240,0.10) 50%,
            rgba(180,200,240,0.04) 75%,
            transparent 100%
          ),
          radial-gradient(ellipse 500px 100px at 50% 50%, rgba(255,235,200,0.16) 0%, transparent 70%),
          radial-gradient(ellipse 350px 80px at 25% 50%, rgba(180,160,220,0.12) 0%, transparent 70%),
          radial-gradient(ellipse 400px 90px at 75% 50%, rgba(160,180,220,0.12) 0%, transparent 70%)
        `,
        backgroundSize: '200% 100%',
        filter: 'blur(10px)',
        animation: 'river-flow 120s linear infinite',
        pointerEvents: 'none',
        willChange: 'transform, background-position',
      }} />

      {/* 星点 — outer parallax wrapper + inner drift wrapper + per-star twinkle */}
      <div style={{
        position: 'absolute', inset: 0,
        transform: parallax(PARALLAX_COEFS.stars),
        willChange: 'transform',
        pointerEvents: 'none',
      }}>
        <div style={{
          position: 'absolute', inset: 0,
          animation: 'stars-drift 360s ease-in-out infinite alternate',
        }}>
          {STARS.map((s, i) => (
            <div key={i} style={{
              position: 'absolute',
              top: `${s.top}%`,
              left: `${s.left}%`,
              width: s.size,
              height: s.size,
              borderRadius: '50%',
              background: '#fff',
              boxShadow: `0 0 ${s.size * 2}px rgba(255,255,255,0.7)`,
              animation: `twinkle ${s.duration}s ease-in-out ${s.delay}s infinite alternate`,
            }} />
          ))}
        </div>
      </div>

      {/* 月亮 — parallax only */}
      <div style={{
        position: 'absolute', top: '8%', right: '6%',
        width: 72, height: 72, borderRadius: '50%',
        background: 'radial-gradient(circle, #f0f4ff 0%, #d8e0f0 40%, rgba(216,224,240,0.2) 70%, transparent 100%)',
        boxShadow: '0 0 60px rgba(216,224,240,0.3)',
        transform: parallax(PARALLAX_COEFS.moon),
        pointerEvents: 'none',
        willChange: 'transform',
      }} />

      {/* 流星 — top decorative layer, no parallax (would look chaotic on fast streaks) */}
      {SHOOTING_STARS.map((s, i) => (
        <div key={i} style={{
          position: 'absolute',
          top: `${s.top}%`,
          left: `${s.left}%`,
          width: s.trailLength,
          height: 1.5,
          background: 'linear-gradient(90deg, transparent 0%, rgba(255,255,255,0.9) 100%)',
          borderRadius: '1px',
          opacity: 0,
          '--angle': `${s.angle}deg`,
          '--dx': `${s.dx}px`,
          '--dy': `${s.dy}px`,
          transform: `rotate(${s.angle}deg)`,
          animation: `shooting-star ${s.duration}s linear ${s.delay}s infinite`,
          pointerEvents: 'none',
        } as React.CSSProperties} />
      ))}
    </>
  );
}
```

**Notes:**
- The `as React.CSSProperties` cast on the shooting star div is required because TypeScript's `CSSProperties` type doesn't allow custom properties (`--angle` etc.) by default
- `parallax(coef)` returns just the `translate(...)` string; each call site composes it with any existing transform (e.g., Milky Way's `translateY(-50%)`)
- Two-wrapper pattern for stars: outer wrapper owns parallax (changes when mouse moves), inner wrapper owns drift (independent infinite animation). Both apply `transform` — putting them on the same element would make the animation overwrite the parallax on every frame
- `pointerEvents: 'none'` on every layer is preserved — these layers must not block the wheel/drag handlers on the underlying canvas

**Background for the engineer:** `scripts/build-standalone.cjs` inlines the React component code as a JS string template (because standalone.html uses Babel-standalone in the browser to compile JSX at runtime — no build step). This means every JSX component is duplicated as a template literal inside `build-standalone.cjs`. The template literals use backtick-escaping (`\\\`` for backticks inside backticks) and `${...}`-escaping (`\\\${...}` for literal `${` inside template literals). When you copy code from a `.tsx` source into the template, you must apply this escaping to any nested template literals.
